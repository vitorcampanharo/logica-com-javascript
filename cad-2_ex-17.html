Ler as notas da 1a. e 2a. avaliações de um aluno. Calcular a média aritmética simples e escrever uma mensagem que diga se o aluno foi ou não aprovado (considerar que nota igual ou maior que 6 o aluno é aprovado). Escrever também a média calculada. 

        
            <label for="nota_1">Digite sua primeira nota:
            <input type="number" name="nota_1" id="nota_1"> <br> <br>
            <label for="nota_2">Digite sua segunda nota:</label>
            <input type="number" name="nota_2" id="nota_2"> <br> <br>
            <button onclick="calcular()">Calcular
            <button onclick="limpar()">Limpar
            <div id="result">
            <script>
                function calcular() {
                    var nota_1 = Number (document.getElementById("nota_1").value)
                    var nota_2 = Number (document.getElementById("nota_2").value)
                    var media = (nota_1+nota_2)/2
                    if (media >= 6) {
                        document.querySelector("#result").innerHTML=`Sua média é ${media}, parabéns, está aprovado`
                    } else{
                        document.querySelector("#result").innerHTML = `Sua média é ${media}, infelizmente está reprovado.`
                    }            
                }
                function limpar() {
                    document.querySelector("#result").innerHTML= null
                    var num = document.getElementById("nota_1").value=""
                    var num = document.getElementById("nota_2").value=""
                }
            </script>
