33) Ler dois valores e imprimir uma das três mensagens a seguir: Números iguais, caso os números sejam iguais Primeiro é maior, caso o primeiro seja maior que o segundo; Segundo maior, caso o segundo seja maior que o primeiro. 

        
            <section>
                <label for="num1">Primeiro número</label>
                <input type="number" name="num1" id="num1">
            </section> <br>
            <section>
                <label for="num2">Segundo número</label>
                <input type="number" name="num2" id="num2">
            </section> <br>
            <button onclick="verificar()">Verificar</button>
            <button onclick="limpar()">Limpar
            <div id="tela"></div>
            <script>
                function verificar() {
                    var num1 = Number (document.getElementById("num1").value)
                    var num2 = Number (document.getElementById("num2").value)
                    var tela = document.querySelector("#tela")
                    if (num1 > num2) {
                        tela.innerHTML = `O primeiro número ${num1} é maior que o segundo número ${num2}`
                    } else if(num1 < num2){
                        tela.innerHTML = `O primeiro número ${num1} é menor que o segundo número ${num2}`
                    } else if(num1 === num2){
                        tela.innerHTML = `Números iguais, ${num1} e ${num2}`
                    }
                }
                function limpar() {
                    tela.innerHTML = null
                    num1.value = null
                    num2.value = null
                }
            </script>
