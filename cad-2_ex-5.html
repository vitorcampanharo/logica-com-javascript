5) Escreva um algoritmo para ler um valor (do teclado) e escrever (na tela) o seu antecessor

        
            <script>
                while (true) {
                    var num = parseInt(prompt("Digite um número:"));
                    if (num<=0) {
                        alert("Digite apenas números acima de zero")
                        continue
                    }
                    if (num>0) {
                        break
                    }
                }
                if (num>0) {
                    num-=1
                }
                alert(num);
            </script>
