Crie um programa em que dois usuários digitem um valor, e seja sorteado um numero entre 0 ou 1. Se for 0, o usuário que digitou menor valor ganha. Se for 1, o usuário que digitou mair valor ganha.

       
        var jogador1 = parseInt(prompt("Jogador 1, digite seu número:"));
        var jogador2 = parseInt(prompt("Jogador 2, digite seu número:"));        
        if (jogador1==jogador2) {
            alert("Empate, jogo encerrado")
        }
        if (jogador1!=jogador2) {
            var sorteio = parseInt(Math.random() * 2);
            if (sorteio==0) {           
                alert("O número sorteado foi "+sorteio);
                if (jogador1>jogador2) {
                    alert("Jogador 2 ganhou");
                }
                if (jogador1 < jogador2) {
                    alert("Jogador 1 ganhou")
                }
            }
            if (sorteio==1) {
                alert("O número sorteado foi "+sorteio);
                if (jogador1>jogador2) {
                    alert("Jogador 1 ganhou")
                }
                if (jogador1 < jogador2) {
                    alert("Jogador 2 ganhou")
                }
            }
        }
