Ler um número e verificar se ele é par ou ímpar. Quando for par armazenar esse valor em P e quando for ímpar armazená-lo em I. Exibir P e I no final do processamento.

        
            <script>
                var num = parseFloat(prompt("Digite um valor:"));
                var P = 0;
                var I = 0;
                if (num%2==0) {
                    P+=num;
                    alert("Numero solicitado foi "+ P + ", portanto é par");
                }
                if (num%2==1) {
                    I+=num;
                    alert("Numero solicitado foi "+ I + ", portanto é impar");
                }
            </script>
