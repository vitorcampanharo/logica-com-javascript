Enunciado: Tendo como dados de entrada o nome, a altura e o sexo (M ou F) de uma pessoa, calcule e mostre seu peso ideal, utilizando as seguintes fórmulas: - para sexo masculino: peso ideal = (72.7 * altura) - 58 - para sexo feminino: peso ideal = (62.1 * altura) - 44.7 

        
            <div>
                <label for="nom">Qual seu nome:</label>
                <input type="text" name="nom" id="nome">
            </div>
            <div>
                <label for="alt">Qual sua altura
                <input type="number" name="alt" id="altura" step="0.01">
            </div>
            <div>
                <label for="sex">Qual seu sexo:</label>
                <input type="radio" name="sex" id="M"> Masculino
                <input type="radio" name="sex" id="F"> Feminino
            </div>
            <div>
                <button onclick="calc()"> Calcular
                <button onclick="limpar()">Limpar
            </div>
            <div id="tela">
            <script>
                function calc() {
                    var nome = document.getElementById("nome").value
                    var altura = Number(document.getElementById("altura").value)
                    var peso_homem = (72.7*altura)-58
                    var peso_mulher = (62.1*altura)-44.7
                    var mas = document.getElementsByName("sex")[0]
                    var fem = document.getElementsByName("sex")[1]
        
                    if (mas.checked) {
                    document.querySelector("#tela").innerHTML = `Olá ${nome}, você é um homem de ${altura} metros, seu peso ideal é de ${peso_homem} kg`
                    } else if(fem.checked){
                    document.querySelector("#tela").innerHTML = `Olá ${nome}, você é uma mulher de ${altura} metros, seu peso ideal é de ${peso_mulher} kg`
                    }
                }
                function limpar() {
                    document.querySelector("#tela").innerHTML= null
                    var num = document.getElementById("nome").value=""
                    var num = document.getElementById("altura").value=""
                    var num = document.getElementById("M").checked=null
                    var num = document.getElementById("F").checked=null
                }
            </script>
