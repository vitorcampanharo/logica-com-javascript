Ler o ano atual e o ano de nascimento de uma pessoa. Escrever uma mensagem que diga se ela poderá ou não votar este ano (não é necessário considerar o mês em que a pessoa nasceu). 

        
            <label for="num">Seu ano de nascimento:
            <input type="number" name="num" id="num" max="2022" min="1922">
            <button onclick="verificar()">Verificar
            <button onclick="limpar()">Limpar
            <div id="result">
            <script>
                function verificar() {
                    var num = Number (document.getElementById("num").value)
                    var idade = (2022 - num)
                    if (idade < 16) {
                        document.querySelector("#result").innerHTML = `Sua idade é de ${idade} anos. Está proibido de votar`
                    } else if (idade < 18) {
                        document.querySelector("#result").innerHTML = `Sua idade é de ${idade} anos. Não precisa votar`
                    } else if (idade >= 70) {
                        document.querySelector("#result").innerHTML = `Sua idade é de ${idade} anos. Seu voto é facultativo`      
                    } else if (idade >= 18) {
                        document.querySelector("#result").innerHTML = `Sua idade é de ${idade} anos. Seu voto é obrigatório`  
                    }
                }
                function limpar() {
                    document.querySelector("#result").innerHTML= null
                    var num = document.getElementById("num").value=""
                }
            </script>
