Mostar na tela todos od dados de um produto específico, contendo quantidade em estoque e preço. Estes dados estarão agrupadps em um objeto.

        
            <div id="mostrar">
            <script>
                var mostrar = document.querySelector("#mostrar");
                var produtos = new Array("caneta", "lapis", "compasso", "borracha")
                var caneta = new Object()
                caneta.nome="caneta"
                caneta.quantidade=10
                caneta.preco=1.50
                var lapis = new Object()
                lapis.nome="lapis"
                lapis.quantidade=15
                lapis.preco=1
                var tela = ""
                tela+="Produtos: "+produtos+ "<br>"
                tela+= "<br>"+"Produto: "+caneta.nome+ "<br>";
                tela+= "Quantidade: "+caneta.quantidade+ "<br>";
                tela+= "Preço: "+caneta.preco+" reais"+ "<br>";
                tela+= "<br>"+"Produto: "+lapis.nome+ "<br>";
                tela+= "Quantidade: "+lapis.quantidade+ "<br>";
                tela+= "Preço: "+lapis.preco+" reais"+ "<br>";
                mostrar.innerHTML=tela
            </script>
