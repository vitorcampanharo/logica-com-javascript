            <div id="tela"></div>
            <script>    
                function verificar() {
                    var time1 = document.getElementById("time1").value
                    var time2 = document.getElementById("time2").value
                    var gols1 = Number (document.getElementById("gols1").value)
                    var gols2 = Number (document.getElementById("gols2").value)
                    var tela = document.querySelector("#tela")
                    if (gols1 > gols2) {
                        tela.innerHTML = `O time ${time1} ganhou, com placar ${gols1} a ${gols2}`
                    } else if(gols2 > gols1){
                        tela.innerHTML = `O time ${time2} ganhou, com placar ${gols2} a ${gols1}`
                    } else {
                        tela.innerHTML = "Houve empate"
                    }
                }
                function limpar() {
                    tela.innerHTML = null
                    time1.value = null
                    time2.value = null
                    gols1.value = null
                    gols2.value = null
                }
            </script>
