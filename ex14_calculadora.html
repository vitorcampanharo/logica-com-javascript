<button onclick="somar()">+</button>
            <button onclick="subtrair()">-</button>
            <button onclick="multiplicar()">*</button>
            <button onclick="dividir()"></button>
            <button class="limpar" onclick="limpar()">Limpar</button>
            <div id="calculadora"></div>
    <script>
        var num_principal = parseFloat(prompt("Digite um número para iniciar o cálculo:"));
        var calc = document.querySelector("#calculadora");
        calc.innerHTML=num_principal
        function somar() {
            var calc = document.querySelector("#calculadora");
            var num_secundario = parseFloat(prompt("Digite o próximo número a ser calculado:"));
            num_principal += num_secundario;
            calc.innerHTML = num_principal;
        }
        function subtrair() {
            var calc = document.querySelector("#calculadora");
            var num_secundario = parseFloat(prompt("Digite o próximo número a ser calculado:"));
            num_principal = num_principal-num_secundario;
            calc.innerHTML = num_principal;
        }
        function multiplicar() {
            var calc = document.querySelector("#calculadora");
            var num_secundario = parseFloat(prompt("Digite o próximo número a ser calculado:"));
            num_principal = num_principal*num_secundario;
            calc.innerHTML = num_principal;
        }
        function dividir() {
            var calc = document.querySelector("#calculadora");
            var num_secundario = parseFloat(prompt("Digite o próximo número a ser calculado:"));
            num_principal = num_principal/num_secundario;
            calc.innerHTML = num_principal;
        }
        function limpar() {
            var calc = document.querySelector("#calculadora");
            calc.innerHTML = "";
        }
    </script>
