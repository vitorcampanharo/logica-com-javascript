Mostrar na tela o dia da semana em que o usuário está, usando if

        
            <script>
                var hoje = new Date();
                var dia_da_semana = hoje.getDay();
                if (dia_da_semana == 0) {
                    alert("Bem vindo, hoje é domingo");
                } else if(dia_da_semana == 1){
                    alert("Bem vindo, hoje é segunda feira");
                } else if (dia_da_semana == 2){
                    alert("Bem vindo, hoje é terça feira")
                } else if (dia_da_semana == 3){
                    alert("Bem vindo, hoje é quarta feira")
                } else if (dia_da_semana == 4){
                    alert("Bem vindo, hoje é quinta feira")
                } else if (dia_da_semana == 5){
                    alert("Bem vindo, hoje é sexta feira")
                } else if (dia_da_semana == 6){
                    alert("Bem vindo, hoje é sábado")
                }
            </script>
