Elabore um algoritmo que gera e escreve os números ímpares dos números lidos entre 100 e 200.

        
            <script>
                var campo = document.querySelector("#campo");
                var campo_calculado = ""
                for(i=100; i<=200; i++){
                    if (i%2==1) {
                        campo_calculado+=i + "<br>";
                    }
                }
                campo.innerHTML=campo_calculado
            </script>
