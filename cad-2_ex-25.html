Faça um algoritmo para ler: número da conta do cliente, saldo, débito e crédito. Após, calcular e escrever o saldo atual (saldo atual = saldo - débito + crédito). Também testar se saldo atual for maior ou igual a zero escrever a mensagem 'Saldo Positivo', senão escrever a mensagem 'Saldo Negativo'. 

        
            <main>
                <section>
                    <label for="conta">Número da conta:</label> <br>
                    <input type="text" name="conta" id="conta">
                </section> <br>
                <section>
                    <label for="saldo">Seu saldo:</label> <br>
                    <input type="number" name="saldo" id="saldo">
                </section> <br>
                <section>
                    <label for="débito">Seu débito:</label> <br>
                    <input type="number" name="débito" id="débito">
                </section> <br>
                <section>
                    <label for="crédito">Seu crédito: <br>
                    <input type="number" name="crédito" id="crédito">
                 <br>
                

                    <button onclick="calcular()">Calcuar saldo atual</button>
                    <button onclick="resetar()">Resetar
                    
Seu saldo atual: <span id="tela"></span></p>
                

            </main>
            <script>
                function calcular() {
                    var conta = Number (document.getElementById("conta").value)
                    var saldo = Number (document.getElementById("saldo").value)
                    var débito = Number (document.getElementById("débito").value)
                    var crédito = Number (document.getElementById("crédito").value)
                    var tela = document.querySelector("#tela")
                    var saldo_atual = saldo-débito+crédito
                    var msg = ""
                    if (saldo_atual >=0) {
                        msg+="Saldo positivo"
                    } else {
                        msg+="Saldo negativo"
                    }
                    tela.innerHTML = `${saldo_atual}. ${msg}`
                }
                function resetar() {
                    conta.value = null
                    saldo.value = null
                    débito.value = null
                    crédito.value = null
                    tela.innerHTML = null
                }
            </script>
