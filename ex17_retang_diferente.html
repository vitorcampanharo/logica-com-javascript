        <p>Área total: <span id="area_total"></span>

        <script>
            function calcular() {
                var largura = document.getElementById("ilargura").value;
                var altura = document.getElementById("ialtura").value;
                var area_total = document.querySelector("#area_total");
                var resultado_area =(largura*altura)+" cm2";
                area_total.innerHTML=resultado_area
                }
                function limpar() {
                var area_total = document.querySelector("#area_total");
                area_total.innerHTML=""
                }
                </script>
