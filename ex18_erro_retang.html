            function calcular_area() {
            var largura = parseFloat(prompt("Qual a largura:"));
            var altura = parseFloat(prompt("Qual a altura:"));
            if (isNaN(largura) || isNaN(altura)) {
            throw new Error("Digite apenas números")
            }
            var resultado = (largura*altura);
            var area_calculada = document.querySelector("#area_calculada");
            area_calculada.innerHTML=resultado
            }
