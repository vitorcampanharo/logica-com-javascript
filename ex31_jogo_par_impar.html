                function jogar() {
                    var num = document.getElementById("number").value;
                    if (num>10 || num<0) {
                        var msg = "";
                        msg+="Apenas números de 0 a 10, por favor";
                        document.querySelector("#tela").innerHTML=msg
                    }
                    if (num <= 10 && num >=0) {
                        if (num%2==0) {
                            var msg = "";
                            msg+="É par";
                            document.querySelector("#tela").innerHTML=msg
                        } else if (num%2==1){
                            var msg = "";
                            msg+="É impar";
                            document.querySelector("#tela").innerHTML=msg
                        }
                    }          
                }
                function apagar() {
                    var msg = ""
                    document.querySelector("#tela").innerHTML=msg
                    var num = document.getElementById("number").value=null;
                }       
            </script>
