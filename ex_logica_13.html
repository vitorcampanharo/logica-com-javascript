Faça um diagrama de bloco que leia um número inteiro e mostre uma mensagem indicando se este número é par ou ímpar, e se é positivo ou negativo.

        
            <script>
                while (true) {
                    var num = parseInt(prompt("Digite um número:"));
                    if (num===0) {
                        alert("Digite um número diferente de 0");
                        continue
                    }
                    if (num!=0) {
                        if (num%2==0 && num>0) {
                        alert("É um número par e positivo");
                    }
                    if (num%2==0 && num<0) {
                        alert("É um número par e negativo");
                    }
                    if (num%2==1 && num>0) {
                        alert("É um número ímpar e positivo");
                    }
                    if (num%2!=0 && num<0) {
                        alert("É um número ímpar e negativo");
                    }   
                        break
                    }    
                } 
            </script>
