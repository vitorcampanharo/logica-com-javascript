Calcular o peso ideal Para homens: (72.7*h) - 58 Para mulheres: (62.1*h) - 44.7

        
            <script>
                var H = parseFloat(prompt("Voçe homem, digite sua altura:"));
                var pesoH = (72.7*H)-58;
                alert("Seu peso ideal é de "+ pesoH+ "Kg");
                var M = parseFloat(prompt("Voçe mulher, digite sua altura:"));
                var pesoM = (62.1*M)-44.7;
                alert("Seu peso ideal é de "+ pesoM+ "Kg");
            </script>
