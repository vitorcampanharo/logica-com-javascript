Mostrar na tela seus dados como nome, sobrenome, cpf, tel, usando loop for in

        
            <script>
                var dados = {
                    Nome: ["Vitor","juju"],
                    Sobrenome: ["Campanharo","Mendonça"],
                    CPF: ["169.194.577-33","459.635.985-85"],
                    TEL: ["21 974072185","21 974582548"]
                }
                var tela = "";
                for(propriedade in dados){
                    tela+=propriedade+": "+dados[propriedade][0] + "<br>";
                }
                tela+="<br>"+"<br>"
                for(propriedade in dados){
                    tela+=propriedade+": "+dados[propriedade][1] + "<br>";
                }
                document.querySelector("#tela").innerHTML=tela
            </script>
