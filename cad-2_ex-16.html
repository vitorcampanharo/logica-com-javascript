As maçãs custam R$ 1,30 cada se forem compradas menos de uma dúzia, e R$ 1,00 se forem compradas pelo menos 12. Escreva um programa que leia o número de maçãs compradas, calcule e escreva o custo total da compra. 

        
            <label for="num">Digite a quantidade de maçãs desejadas:
            <input type="number" name="num" id="num">
            <button onclick="verificar()">Verificar
            <button onclick="limpar()">Limpar
            <div id="result">
            <script>
                function verificar() {
                    var num = document.getElementById("num").value
                    if (num < 12) {
                        num*=1.30
                        document.querySelector("#result").innerHTML=`Comprando abaixo de 12 você paga R$1,30 em cada. Seu total é de R$${num} reais.`
                    } else if (num >= 12) {
                        num*=1
                        document.querySelector("#result").innerHTML=`Comprando abaixo de 12 você paga R$1 em cada. Seu total é de R$${num} reais.`
                    }
                }
                function limpar() {
                    document.querySelector("#result").innerHTML= null
                    var num = document.getElementById("num").value=""
                }
            </script>
