Faça um algoritmo para ler as 3 notas obtidas por um aluno nas 3 verificações e a média dos exercícios que fazem parte da avaliação. Calcular a média de aproveitamento, usando a fórmula abaixo e escrever o conceito do aluno de acordo com a tabela de conceitos mais abaixo: 

            
                <section>
                    <label for="nota1"></label>
                    <input type="number" name="nota1" id="nota1" placeholder="Nota 1">
                </section>
                <section>
                    <label for="nota2"></label>
                    <input type="number" name="nota2" id="nota2" placeholder="Nota 2">
                </section>
                <section>
                    <label for="nota3"></label>
                    <input type="number" name="nota3" id="nota3" placeholder="Nota 3">
                </section>
                <button onclick="calcular()">Calcular</button>
                <div id="tela"></div>
                <script>
                    function calcular() {
                        let nota1 = Number (document.getElementById("nota1").value)
                        let nota2 = Number (document.getElementById("nota2").value)
                        let nota3 = Number (document.getElementById("nota3").value)
                        let mediaExercicio = (nota1 + nota2 + nota3) / 3
                        let mediaAproveitamento = (nota1 + (nota2*2) + (nota3*3) + mediaExercicio) / 7
                        let tela = document.getElementById("tela")
                        if (mediaAproveitamento < 6) {
                            tela.innerHTML = `Sua média é ${mediaAproveitamento.toFixed(2)}, logo seu conceito é D`
                        } else if (mediaAproveitamento<7.5) {
                            tela.innerHTML = `Sua média é ${mediaAproveitamento.toFixed(2)}, logo seu conceito é C`
                        } else if (mediaAproveitamento<9) {
                            tela.innerHTML = `Sua média é ${mediaAproveitamento.toFixed(2)}, logo seu conceito é B`
                        } else{
                            tela.innerHTML = `Sua média é ${mediaAproveitamento.toFixed(2)}, logo seu conceito é A`
                        }
                        console.log(mediaExercicio)
                        console.log(mediaAproveitamento.toFixed(2))
                    }
