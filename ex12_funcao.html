Pedir para o usuário digitar um número e mostrar a tabuada desse número de 1 a 1000. Deve ter a possibilidade do usuário digitar outro número e também limpar a tabuada gerada previamente.

        
            function mostrarTabuada() {
                var espaço = document.querySelector('#espaço');
                var mensagem = ''
                var numero1 = parseInt(prompt("Qual número deseja calcular:"));
                var numero2 = 0
                while (numero2 < 1000) {
                    numero2 = numero2+1;
                    mensagem += numero1+ ' * '+ numero2 +' = '+ (numero1*numero2)+ '< br>'
                }
                espaço.innerHTML = mensagem
            }
            function limparTabuada() {
                mensagem = ""
                espaço.innerHTML = mensagem
            }
