Pedir para o usuário digitar dois valores e mostrar uma mensagem informando se foi aprovado ou não. A nota de corte é 5.

        
            var num_1 = parseFloat(prompt("Nota de português:"));
            var num_2 = parseFloat(prompt("Nota de matemática:"));
            var media = (num_1+num_2)/2;
            if (media>=5) {
                alert("Parabens, você foi aprovado")
            } else {
                alert("Que pena, você foi reprovado");
            }
