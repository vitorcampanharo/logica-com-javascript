leia 4 números; calcule o quadrado de cada um; somem todos; mostre o resultado.

        
            <script>
                var valor1 = parseFloat(prompt("Digite o primeiro valor:"));
                var valor2 = parseFloat(prompt("Digite o segundo valor:"));
                var valor3 = parseFloat(prompt("Digite o terceiro valor:"));
                var valor4 = parseFloat(prompt("Digite o quarto valor:"));
                var quad_valor1 = valor1*valor1;
                var quad_valor2 = valor2*valor2;
                var quad_valor3 = valor3*valor3;
                var quad_valor4 = valor4*valor4;
                var soma_todos = quad_valor1+quad_valor2+quad_valor3+quad_valor4;
                alert("O resultado final é "+ soma_todos);
            </script>
