Mostrar na tela o dia da semana em que o usuário está acessando, usando Array

        
            <script>
                var semana = new Array("domingo", "segunda", "terça", "quarta", "quinta","sexta", "sábado");
                var hoje = new Date();
                var dia_da_semana = hoje.getDay();
                alert("Seja bem vindo, hoje é "+semana[dia_da_semana]);
            </script>
