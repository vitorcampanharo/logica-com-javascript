Escreva um algoritmo para ler o salário mensal atual de um funcionário e o percentual de reajuste. Calcular e escrever o valor do novo salário. 

        
            <script>
                var salario = parseFloat(prompt("Qual o salário atual:"));
                var reajuste = parseFloat(prompt("Qual será o reajuste em % desse mês"));
                var salario_final = (reajuste/100)*salario+salario
                
                alert(`O salário atual é de ${salario.toLocaleString("pt-BR", {style:"currency", currency:"BRL"})} e o reajuste deste mês será de ${reajuste}%; o salário final será ${salario_final.toLocaleString("pt-BR", {style: "currency", currency: "BRL"})}`);
            </script>
