Ler um valor e escrever a mensagem É MAIOR QUE 10! se o valor lido for maior que 10, caso contrário escrever NÃO É MAIOR QUE 10! 

        
        <label for="num">Digite um número:
            <input type="number" name="num" id="num">
            <button onclick="verificar()">Verificar
            <div id="result">
            <script>
                function verificar() {
                    var num = document.getElementById("num").value
                    if (num > 10) {
                        document.querySelector("#result").innerHTML="É MAIR QUE 10"
                    } else if(num < 10){
                        document.querySelector("#result").innerHTML="NÃO É MAIOR QUE 10"
                    } else if (num == 10) {
                        document.querySelector("#result").innerHTML="RESULTADO IMPROVÁVEL"
                    }
                }
            </script>
