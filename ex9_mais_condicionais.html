Crie um programa em que o usuário digite duas notas, e a mensagem se for aprovado, a nota de corte é 5. Porém, se a nota for 0 em qualquer uma das matérias, reprova automaticamente.

        
            var nota1 = parseFloat(prompt("Nota de português"));
            var nota2 = parseFloat(prompt("Nota de matemática"));       
            if (nota1==0 || nota2==0) {
                alert("Nenhuma de suas notas pode ser 0, portanto está automaticamente reprovado")
            }
            if (nota1!=0 && nota2!=0) {
                var media = (nota1+nota2)/2;
                if (media>=5) {
                    alert("Aprovado")
                } else {
                    alert("Reprovado")
                }
            }
